void processLayer(const vector& inputArr, const vector>& weightsArr,
    const vector& biasArr, vector& resultArr,
    int inputSize, int outputSize, bool applySoftmax) {
    vector sums(outputSize, 0.0f);

    for (int i = 0; i < outputSize; ++i) {
        sums[i] = biasArr[i];

        for (int j = 0; j < inputSize; ++j) {
            sums[i] += inputArr[j] * weightsArr[j][i];
        }
    }

    if (applySoftmax) {

        float maxVal = *max_element(sums.begin(), sums.end());
        float total = 0.0f;

        for (float& val : sums) {
            val = exp(val - maxVal);
            total += val;
        }
        
        for (float& val : sums) {
            val /= total;
        }
    }
    else {
        for (float& val : sums) {
            val = ReLU(val);
        }
    }
    resultArr = sums;
}

#include <iostream>

double calculateTemperature(double energy, double particles) {
    return energy / particles;
}

ðŸŒŒ Theory of Relativity in 3D

A 3D simulation visualizing time dilation and length contraction. 


#include <iostream>
#include <cmath>

double lorentzFactor(double velocity, double c) {
    return 1 / sqrt(1 - (velocity * velocity) / (c * c));
}

int main() {
    double gamma = lorentzFactor(0.9, 1.0);
    std::cout << "Lorentz Factor: " << gamma << std::endl;
    return 0;
}
